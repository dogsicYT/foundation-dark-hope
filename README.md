// Инициализация
function init() {
    populateArcSelect();
    arcSelect.addEventListener('change', onArcChange);
    window.addEventListener('hashchange', onHashChange);
    updatePageFromHash();
    renderChaptersList(); // Добавлено: рендерим список глав при инициализации
}

// Заполнение выпадающего списка арок
function populateArcSelect() {
    arcSelect.innerHTML = '';
    Object.keys(arcsData).forEach(arcId => {
        const option = document.createElement('option');
        option.value = arcId;
        option.textContent = arcsData[arcId].name;
        arcSelect.appendChild(option);
    });
    arcSelect.value = currentArcId;
}

// Обработка смены арки
function onArcChange() {
    currentArcId = arcSelect.value;
    currentChapterId = null;
    renderChaptersList();
    showDefaultView();
    window.location.hash = '';
}

// Рендер списка глав - ИСПРАВЛЕННАЯ ВЕРСИЯ
function renderChaptersList() {
    chaptersList.innerHTML = '';
    const selectedArc = arcsData[currentArcId];
    if(!selectedArc) return;
    
    currentArcName.textContent = selectedArc.name;
    
    selectedArc.chapters.forEach(ch => {
        const li = document.createElement('li');
        li.className = 'chapter-item';
        li.dataset.chapterId = ch.id;
        
        // Создаем содержимое элемента списка
        li.innerHTML = `
            <div style="display: flex; align-items: center; width: 100%;">
                <span class="chapter-number">${ch.id}</span>
                <span class="chapter-title">${ch.title.replace(`Глава ${ch.id}: `, '')}</span>
            </div>
        `;
        
        // Обработчик клика
        li.addEventListener('click', (e) => {
            e.preventDefault();
            selectChapter(ch.id);
        });
        
        chaptersList.appendChild(li);
    });
}

// Обработка хэша
function onHashChange() {
    updatePageFromHash();
}

function updatePageFromHash() {
    const hash = window.location.hash;
    if(!hash || !hash.startsWith('#chapter-')) {
        showDefaultView();
        return;
    }
    const chapterIdStr = hash.replace('#chapter-', '');
    const chapterId = parseInt(chapterIdStr);
    const selectedArc = arcsData[currentArcId];
    if(!selectedArc) return;
    const chapter = selectedArc.chapters.find(c => c.id === chapterId);
    if(!chapter) {
        showDefaultView();
        return;
    }
    showChapter(chapter);
    highlightActiveChapter(chapterId);
}

function highlightActiveChapter(chapterId) {
    Array.from(chaptersList.children).forEach(li => {
        li.classList.remove('active');
        if (parseInt(li.dataset.chapterId) === chapterId) {
            li.classList.add('active');
        }
    });
}

function showDefaultView() {
    chapterTitle.textContent = 'Выберите главу для чтения';
    chapterSubtitle.textContent = 'Нажмите на любую главу в списке слева, чтобы начать чтение';
    chapterDate.textContent = '';
    chapterContent.innerHTML = '<p>Выберите главу из списка слева, чтобы отобразить её содержимое здесь.</p>';
    currentChapterId = null;
    
    Array.from(chaptersList.children).forEach(li => {
        li.classList.remove('active');
    });
}

function showChapter(chapter) {
    chapterTitle.textContent = chapter.title;
    chapterSubtitle.textContent = chapter.subtitle;
    chapterDate.textContent = chapter.date;
    chapterContent.innerHTML = chapter.content;
    currentChapterId = chapter.id;
    
    chapterContent.scrollTop = 0;
}

function selectChapter(chapterId) {
    window.location.hash = `#chapter-${chapterId}`;
}
